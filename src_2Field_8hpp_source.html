40  /* Default constructor, builds a field from some default value
41  * (i.e. the context for NTL).
42  *
43  * If test is false, do not perform primality and irreducibility
44  * tests
45  *
46  * throws : NotPrimeException, NotIrreducibleException if the
47  * default makes no sense
48  */
49  template <class T> const Field<T>& Field<T>::createField(const bool test)
50  throw (NotPrimeException, NotIrreducibleException) {
51  // retrieve all the informations on the modulus
52  BigInt p = GFp::modulus();
53  const GFpXModulus& P = GFpE::modulus();
54  long d = deg(P);
55  Context context; context.p.save(); context.P.save();
56  // test primality
57 #ifdef FAAST_TIMINGS
58  TIME.PRIMETEST = -GetTime();
59 #endif
60  if ( p <= long(1) || (test && !ProbPrime(p)) ) {
61  throw NotPrimeException();
62  }
63 #ifdef FAAST_TIMINGS
64  TIME.PRIMETEST += GetTime();
65 #endif
66  // test irreducibility
67 #ifdef FAAST_TIMINGS
68  TIME.IRREDTEST = -GetTime();
69 #endif
70  if (d > 1 && test && !DetIrredTest(P)) {
71  throw NotIrreducibleException();
72  }
73 #ifdef FAAST_TIMINGS
74  TIME.IRREDTEST += GetTime();
75 #endif
76  // build GF(p^d)
77  if (d >= 2) {
78  // build GF(p)
79  GFp one; one = 1;
80  Field<T>* baseField = new Field<T>(context, one, p);
81  // compute the generator of this field
82  GFpE primitive; conv(primitive, GFpX(1,1));
83  // build the field
84  Field<T>* K = new Field<T>(baseField, context, primitive,
85  p, d, primitive);
86  // connect the base field
87  baseField->overfield = K;
88  return *K;
89  }
90  // build GF(p)
91  else {
92  GFp primitive; primitive = 1;
93  return *(new Field<T>(context, primitive, p));
94  }
95  }

154  /* Build the field GF(p^d) using some default
155  * polynomial.
156  * Notice that this operation implicitely creates
157  * the field GF(p) too.
158  *
159  * If test is false, do not perform primality and irreducibility
160  * tests
161  *
162  * throws : NotPrimeException if p is not prime
163  * throws : FAASTException if d less than one
164  */
165  template <class T> const Field<T>& Field<T>::createField
166  (const BigInt& p, const long d, const bool test)
167  throw (NotPrimeException, BadParametersException) {
168  if (d < 1) {
169  throw BadParametersException("Cannot create an extension field of negative degree.");
170  }
171  if (p <= long(1)) {
172  throw NotPrimeException();
173  }
174 #ifdef FAAST_TIMINGS
175  TIME.PRIMETEST = -GetTime();
176 #endif
177  if (test && !ProbPrime(p)) {
178  throw NotPrimeException();
179  }
180 #ifdef FAAST_TIMINGS
181  TIME.PRIMETEST += GetTime();
182 #endif
183 
184  GFp::init(p);
185  GFpX P;
186 #ifdef FAAST_TIMINGS
187  TIME.BUILDIRRED = -GetTime();
188 #endif
189  if (d >= 2) BuildIrred(P, d);
190  else SetX(P);
191 #ifdef FAAST_TIMINGS
192  TIME.BUILDIRRED += GetTime();
193 #endif
194  return createField(P, false);
195  }

228  /* The minimal polynomial over the immediate subfield of the
229  * element returned by generator.
230  */
231  template <class T> FieldPolynomial<T> Field<T>::generatingPolynomial()
232  const throw() {
233  // if this is GF(p) or a base field, this is the same as
234  // the primitive polynomial
235  if (height == 0) return primitivePolynomial();
236  // else, it is X^p - X - alpha
237  else {
238  switchContext();
239  FieldPolynomial<T> res(-(*alpha));
240  res.setCoeff(1, -1);
241  res.setCoeff(to_long(p));
242  return res;
243  }
244  }

246  /* The minimal polynomial over GF(p) of the element returned
247  * by primitiveElement().
248  */
249  template <class T> FieldPolynomial<T> Field<T>::primitivePolynomial()
250  const throw() {
251  switchContext();
252  // if this is GF(p), return X-1
253  if (d == 1) {
254  FieldPolynomial<T> res = -one();
255  res.setCoeff(1);
256  return res;
257  }
258  // else, return the modulus
259  else return FieldPolynomial<T>(this, GFpE::modulus().val());
260  }

332 /****************** Field lattice navigation ******************/
333  /* The field GF(p) */
334  template <class T> const Field<T>& Field<T>::primeField() const throw() {
335  if (!stem->subfield) return *stem;
336  const Field<T>* result = stem->subfield;
337  while (result->subfield) result = result->subfield;
338  return *result;
339  }
340  /* The base field of the tower */
341  template <class T> const Field<T>& Field<T>::baseField() const throw() {
342  const Field<T>* result = stem;
343  while (result->height != 0) result = result->subfield;
344  return *result;
345  }

347 /****************** Comparison ******************/
348  /* There is inclusion between two fields only if the inclusion
349  * has actually been computed.
350  */
351  template <class T> bool Field<T>::isSubFieldOf(const Field<T>& f)
352  const throw () {
353  const Field<T>* stemf = f.stem;
354 #ifdef FAAST_DEBUG
355  if (!stemf || !stem) throw FAASTException("Error : Stem is NULL.");
356 #endif
357  while (stemf != stem && (stemf = stemf->subfield));
358  return stemf == stem;
359  }
