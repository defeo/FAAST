30  /* n-th iterated frobenius. This algorithm is detailed in the
31  * long version.
32  */
33  template <class T> void
34  FieldElement<T>::self_frobenius(long n) throw() {
35  if (isZero()) return;
36  if (base) return;
37 
38  n %= parent_field->d;
39  if (n < 0) n = parent_field->d + n;
40  if (n == 0) return;
41  long p = parent_field->p;
42  long smalld = parent_field->baseField().d;
43  // The small part
44  const long r = n % smalld;
45  SmallFrob(r);
46  // The big part
47  n /= smalld;
48  long j = 0;
49  while (n != 0) {
50  const long c = n % p;
51  for (long i = 0 ; i < c ; i++) BigFrob(j);
52  n /= p ; j++;
53  }
54  }

56  template <class T> void
57  FieldElement<T>::self_trace(const Field<T>& F)
58  throw(NotASubFieldException) {
59  // zero stays zero
60  if (!parent_field) {
61  *this = F.zero();
62  }
63  // handle errors
64  else if (!F.isSubFieldOf(*parent_field)) {
65  throw NotASubFieldException();
66  }
67  // if F is a prime field, this is the absolute trace
68  else if (F.isPrimeField()) {
69  self_trace();
70  }
71  // else, the trace is minus the coefficient of x_i^(p-1)
72  else {
73  // move to the stem
74  parent_field = parent_field->stem;
75 
76  vector<FieldElement<T> > down;
77  while (parent_field != F.stem) {
78  FAAST::pushDown(*this, down);
79  if (down.size() == unsigned(parent_field->p)) {
80  *this = down[long(parent_field->p) - 1];
81  negate();
82  } else *this = F.stem->zero();
83  }
84  // move out of the stem
85  parent_field = &F;
86  }
87  }

89  template <class T> void
90  FieldElement<T>::self_pseudotrace(unsigned long n) throw() {
91  if (isZero() || n == 1) return;
92  if (n == 0) *this = parent_field->zero();
93  if (base) {
94  *this *= parent_field->scalar(n);
95  return;
96  }
97 
98  // first remove the part above d
99  FieldElement<T> traces;
100  if (n >= unsigned(parent_field->d)) {
101  traces = trace();
102  traces *= parent_field->scalar(n / parent_field->d);
103  }
104  // now go under d
105  n %= parent_field->d;
106 
107  long p = parent_field->p;
108  long smalld = parent_field->baseField().d;
109  // The small part
110  const long r = n % smalld;
111  // The big part
112  n /= smalld;
113  const long j = NumPits(p, n);
114  if (j > 0) {
115  vector<FieldElement<T> > v;
116  BigPTraceVector(v, j-1);
117  long i;
118  for (i = j-1 ; i >= 0 && n != 0 ; i--) {
119  const long c = n / power_long(p, i);
120  if (c == 0) {
121  v[i] = 0;
122  } else if (c > 1) {
123  FieldElement<T> t = v[i];
124  for (long h = 1 ; h < c ; h++) {
125  t.BigFrob(i);
126  v[i] += t;
127  }
128  }
129  if (i < j-1) {
130  for (long h = 0 ; h < c ; h++)
131  v[i+1].BigFrob(i);
132  v[i] += v[i+1];
133  }
134  n %= power_long(p, i);
135  }
136  v[i+1].SmallFrob(r);
137  this->SmallPTrace(r);
138  *this += v[i+1];
139  } else {
140  SmallPTrace(r);
141  }
142  // add the part above d, if needed
143  *this += traces;
144  }

149  /* p^j-th iterated frobenius */
150  template <class T> void FieldElement<T>::BigFrob(const long j) {
151 #ifdef FAAST_DEBUG
152  if (j < 0 || j >= parent_field->height) {
153  stringstream msg;
154  msg << "Bad input to BigFrob : " << j << ".";
155  throw FAASTException(msg.str().c_str());
156  }
157 #endif
158  if (isScalar()) return;
159 
160  BigInt p = parent_field->p;
161  // step 2
162  vector<FieldElement<T> > down;
163  FAAST::pushDown(*this, down);
164  down.resize(p);
165  // step 3
166  if (j < parent_field->height - 1) {
167  for (BigInt i = 0 ; i < p ; i++)
168  down[i].BigFrob(j);
169  }
170  // step 5
171  const FieldElement<T>& beta = parent_field->getPseudotrace(j);
172  vector<FieldElement<T> > result;
173  result.resize(p);
174  for (BigInt i = 0 ; i < p ; i++) {
175  result[i] = down[long(p)-1];
176  for (long j = long(i) - 1 ; j >= 0 ; j--) {
177  result[j] *= beta;
178  if (j > 0) result[j] += result[j-1];
179  else result[j] += down[long(p)-long(i)-1];
180  }
181  }
182  // step 6
183  const Field<T>* parent = parent_field;
184  FAAST::liftUp(result, *this);
185  parent_field = parent;
186  }

189  template <class T> void FieldElement<T>::SmallFrob(const long n) {
190 #ifdef FAAST_DEBUG

192  stringstream msg;

194  throw FAASTException(msg.str().c_str());

197  if (isScalar()) return;
198 
199  for (long i = 0 ; i < n ; i++)
200  self_frobenius();

203  /* p^j-th pseudotrace */
204  template <class T> void FieldElement<T>::BigPTrace(const long j) {
205 #ifdef FAAST_DEBUG
206  if (j < 0 || j >= parent_field->height) {
207  stringstream msg;
208  msg << "Bad input to BigPTrace : " << j << ".";
209  throw FAASTException(msg.str().c_str());
210  }
211 #endif
212  if (isZero()) return;
213 
214  SmallPTrace(parent_field->baseField().d);
215  for (long i = 1 ; i <= j ; i++) {
216  FieldElement<T> t = *this;
217  for (BigInt h = 1 ; h < parent_field->p ; h++) {
218  t.BigFrob(i-1);
219  *this += t;
220  }
221  }
222  }

224  /* Put in the vector v all the p^id pseudotraces for 0 <= i <= j */
225  template <class T> void
226  FieldElement<T>::BigPTraceVector(vector<FieldElement<T> >& v,
227  const long j) const {
228 #ifdef FAAST_DEBUG
229  if (j < 0 || j >= parent_field->height) {
230  stringstream msg;
231  msg << "Bad input to BigPTraceVector : " << j << ".";
232  throw FAASTException(msg.str().c_str());
233  }
234 #endif
235  v.clear(); v.resize(j+1);
236  v[0] = *this;
237  v[0].SmallPTrace(parent_field->baseField().d);
238  for (long i = 1 ; i <= j ; i++) {
239  v[i] = v[i-1];
240  FieldElement<T> t = v[i];
241  for (BigInt h = 1 ; h < parent_field->p ; h++) {
242  t.BigFrob(i-1);
243  v[i] += t;
244  }
245  }
246  }

248  /* n-th pseudotrace, n < d */
249  template <class T> void FieldElement<T>::SmallPTrace(const long n) {
250 #ifdef FAAST_DEBUG
251  if (n < 0 || n > parent_field->baseField().d) {
252  stringstream msg;
253  msg << "Bad input to SmallPTrace : " << n << ".";
254  throw FAASTException(msg.str().c_str());
255  }
256 #endif
257  if (isZero()) return;
258 
259  if (n == 0) {
260  *this = parent_field->zero();
261  } else {
262  FieldElement<T> t(*this);
263  for (long i = 1 ; i < n ; i++) {
264  t.self_frobenius();
265  *this += t;
266  }
267  }
