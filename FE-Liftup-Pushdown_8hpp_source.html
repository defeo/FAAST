43  // The routine MulMod from Section 4
44  template <class T> void MulMod(
45  vector<typename T::GFpX>& W, const long n,
46  const typename T::BigInt& p) {
47  typedef typename T::GFpX GFpX;
48  typedef typename T::BigInt BigInt;
49 
50  GFpX Lead = W[long(p)-1];
51  for (BigInt i = p-long(1) ; i >= long(0) ; i--) {
52  GFpX tmp = W[i];
53  W[i] = 0;
54  long shift = 1;
55  for (long j = 0 ; j < n ; j++) {
56  ShiftAdd(W[i], tmp, shift);
57  shift *= p;
58  }
59  if (i > long(0)) W[i] += W[long(i)-1];
60  }
61  W[1] += Lead;
62  ShiftAdd(W[0], Lead, 1);
63  }

65  // The routine MulMod* from Section 4
66  template <class T> void TransMulMod(
67  vector<typename T::GFpX>& W, const long n,
68  const typename T::BigInt& p) {
69  typedef typename T::GFpX GFpX;
70  typedef typename T::BigInt BigInt;
71 
72  GFpX Lead; RightShift(Lead, W[0], 1);
73  Lead += W[1];
74  for (BigInt i = 0 ; i <= p-long(1) ; i++) {
75  if (i > long(0)) W[long(i)-1] += W[i];
76  GFpX tmp;
77  long shift = 1;
78  for (long j = 0 ; j < n ; j++) {
79  tmp += RightShift(W[i], shift);
80  shift *= p;
81  }
82  W[i] = tmp;
83  }
84  W[long(p)-1] += Lead;
85  }

87  // The routine Push-down-rec from Section 4
88  template <class T> void pushDownRec(
89  const typename T::GFpX& V, long start, long end,
90  vector<typename T::GFpX>& W, const typename T::BigInt& p) {
91  typedef typename T::GFpX GFpX;
92  typedef typename T::BigInt BigInt;
93 
94  long degree = end - start;
95  long k = NumPits(p, degree);
96  // if deg(V) >= p, cut in p slices and apply recursively
97  if (k > 1) {
98  vector<GFpX> Wtmp;
99  W.clear(); W.resize(p);
100  long splitdegree = power_long(p, k-1);
101  for (long i = start + splitdegree * (degree / splitdegree) ; i >= start ; i -= splitdegree) {
102  pushDownRec<T>(V, i, min(i+splitdegree-1, end), Wtmp, p);
103  MulMod<T>(W, k-1, p);
104  for (BigInt j = 0 ; j < p ; j++) W[j] += Wtmp[j];
105  }
106  }
107  // if deg(V) < p, then
108  // V mod Z^p - Z - T
109  // is V
110  else {
111  W.resize(p);
112  for (long i = start ; i <= end ; i++)
113  W[i-start] = coeff(V, i);
114  }
115  }

117  // The routine Push-down-rec* from Section 4
118  template <class T> void TransPushDownRec(
119  vector<typename T::GFpX>& W, typename T::GFpX& V,
120  long start, long end, const typename T::BigInt& p) {
121  typedef typename T::GFpX GFpX;
122  typedef typename T::BigInt BigInt;
123 
124  long degree = end - start;
125  long k = NumPits(p, degree);
126  // if deg(V) >= p, cut in p slices and apply recursively
127  if (k > 1) {
128  vector<GFpX> Wtmp; Wtmp.resize(p);
129  long splitdegree = power_long(p, k-1);
130  SetCoeff(V, end); // hack
131  for (long i = start ; i <= end ; i += splitdegree) {
132  for (BigInt j = 0 ; j < p ; j++)
133  Wtmp[j] = trunc(W[j], splitdegree/p);
134  TransMulMod<T>(W, k-1, p);
135  TransPushDownRec<T>(Wtmp, V, i, min(i+splitdegree-1, end), p);
136  }
137  }
138  else {
139  for (long i = end ; i >= start ; i--)
140  SetCoeff(V, i, coeff(W[i-start], 0));
141  }
142  }

144  /* The routine Transposed mul from Section 4
145  * (step 2 of Lift-up)
146  * Computes the transposed multiplication of the linear form
147  * (0, ..., 0, -form)
148  * by the element
149  * (W[0], ..., W[p-1])
150  * modulo Q
151  */
152  template <class T> void TransposedMul(
153  vector<typename T::GFpX>& W, const typename T::GFpXModulus& Q,
154  const typename T::GFpX& form, const typename T::BigInt& p) {
155  typedef typename T::GFpX GFpX;
156  typedef typename T::GFpXMultiplier GFpXMultiplier;
157  typedef typename T::BigInt BigInt;
158 
159  vector<GFpXMultiplier> Trans; Trans.resize(p);
160 
161  build(Trans[long(p)-1], W[0] + W[long(p)-1], Q);
162  for (BigInt i = 1; i < p ; i++)
163  build(Trans[long(p)-long(i)-1], W[i], Q);
164 
165  GFpX formtmp = -form;
166  for (BigInt i = 0 ; i < p ; i++)
167  TransMulMod(W[i], formtmp, Trans[i], Q);
168  }

170  template <class T> void TransMod(typename T::GFpX& W,
171  const typename T::GFpXModulus& Q, const typename T::BigInt& p) {
172  typedef typename T::GFpX GFpX;
173  typedef typename T::GFpXMultiplier GFpXMultiplier;
174  typedef typename T::BigInt BigInt;
175 
176  long d = deg(Q);
177  // Xn = X^d mod Q
178  GFpX Xn = -Q; SetCoeff(Xn, d, 0);
179  GFpXMultiplier Trans; build(Trans, Xn, Q);
180 
181  GFpX tmp1 = W;
182  GFpX tmp2;
183  long shift = 0;
184  for (BigInt i = 1 ; i < 2*long(p)-1 ; i++) {
185  TransMulMod(tmp2, tmp1, Trans, Q);
186  tmp1 = tmp2;
187  shift += d;
188  ShiftAdd(W, tmp1, shift);
189  }
190  }

194  /* Push the element e down along the stem and store
195  * the result in v.
196  *
197  * throw : NoSubFieldException if e belongs to GF(p)
198  */
199  template <class T>
200  void pushDown(const FieldElement<T>& e, vector<FieldElement<T> >& v)
201  throw(NoSubFieldException) {
202  typedef typename T::GFp GFp;
203  typedef typename T::GFpX GFpX;
204  typedef typename T::BigInt BigInt;
205 
206  if (!e.parent_field) {
207  v.clear();
208  return;
209  }
210  const Field<T>* parent = e.parent_field->stem;
211  if (!parent->subfield)
212  throw NoSubFieldException();
213 
214  parent->switchContext();
215 
216  // if the subfield is prime
217  // simply return the list of coefficients
218  if (parent->subfield->d == 1) {
219  v.resize(deg(rep(e.repExt)) + 1);
220  const Field<T>* base = &(parent->primeField());
221  const GFpX& eX = rep(e.repExt);
222  for (long i = 0 ; i <= deg(eX) ; i++) {
223  v[i].base = true;
224  v[i].repExt = 0;
225  v[i].repBase = coeff(eX, i);
226  v[i].parent_field = base;
227  }
228  }
229  // if the element is a scalar
230  else if (deg(rep(e.repExt)) <= 0) {
231  v.resize(1);
232  v[0].base = false;
233  v[0].repBase = 0;
234  GFp ec = coeff(rep(e.repExt), 0);
235  parent->subfield->switchContext();
236  v[0].repExt = ec;
237  v[0].parent_field = parent->subfield;
238  }
239  // the real push-down algorithm from Section 4
240  else {
241  vector<GFpX> W;
242  BigInt p = parent->p;
243  pushDownRec<T>(rep(e.repExt), 0, deg(rep(e.repExt)), W, p);
244 #if FAAST_DEBUG >= 2
245  for (BigInt i = 0 ; i < p ; i++) {
246  if (deg(W[i]) * p > deg(rep(e.repExt)))
247  throw FAASTException("Problem in pushDownRec.");
248  }
249 #endif
250  // if this extension was built modulo
251  // X^p - X - x0 - 1
252  // the result lies in GF(p)[x0+1].
253  // This brings the elements back to GF(p)[x0]
254  if (parent->plusone) {
255  GFpX xplusone;
256  SetCoeff(xplusone, 1); SetCoeff(xplusone, 0);
257  for (BigInt i = 0 ; i < p ; i++)
258  compose<T>(W[i], W[i], xplusone, p);
259  }
260  // if this extension was built modulo
261  // X^p - X - xi^(2p-1)
262  if (parent->twopminusone) {
263  for (BigInt i = 0 ; i < p ; i++)
264  expand<T>(W[i], W[i], 2*long(p) - 1);
265  }
266 
267  // prepare to work in the subfield
268  parent->subfield->switchContext();
269 
270  // convert the result of push-down-rec to elements
271  // of the subfield
272  v.resize(p);
273  bool base = parent->subfield->d == 1;
274  for (BigInt i = 0 ; i < p ; i++) {
275  v[i].base = base;
276  // this automatically reduces modulo
277  // the defining polynomial if needed
278  if (base) {
279  v[i].repExt = 0;
280  v[i].repBase = coeff(W[i], 0);
281  } else {
282  v[i].repBase = 0;
283  conv(v[i].repExt, W[i]);
284  }
285  v[i].parent_field = parent->subfield;
286  }
287  }
288  }

290  /* Lift the elements in v up along the stem and store
291  * the result in e.
292  * If v is too short, it is filled with zeros.
293  * If v is too long, the unnecessary elements are ignored.
294  *
295  * throw : NotInSameFieldException if the elements of v do not
296  * belong all to the same field.
297  * throw : NoOverFieldException if there's no extension to lift
298  * up to.
299  */
300  template <class T>
301  void liftUp(const vector<FieldElement<T> >& v, FieldElement<T>& e)
302  throw(NotInSameFieldException, NoOverFieldException) {
303  typedef typename T::GFp GFp;
304  typedef typename T::GFpX GFpX;
305  typedef typename T::GFpE GFpE;
306  typedef typename T::GFpXModulus GFpXModulus;
307  typedef typename T::BigInt BigInt;
308 
309  // if v is empty, return the generic 0
310  typename vector<FieldElement<T> >::const_iterator it = v.begin();
311  if (it == v.end()) {
312  e = FieldElement<T>();
313  return;
314  }
315  // check all elements of v belong to the same field
316  // and save this field into parent.
317  // In parallel, check if v represents a scalar
318  const Field<T>* parent = it->parent_field;
319  bool scalar = it->base || deg(rep(it->repExt)) <= 0;
320  for (it++ ; it != v.end() ; it++) {
321  if (!parent) parent = it->parent_field;
322  else if (it->parent_field && it->parent_field != parent)
323  throw NotInSameFieldException();
324  scalar = scalar && *it == 0;
325  }
326  // standard checks
327  if (!parent) {
328  e = FieldElement<T>();
329  return;
330  }
331  parent = parent->stem;
332  if (!parent->overfield) throw NoOverFieldException();
333 
334  parent->switchContext();
335 
336  // if this is a prime field
337  // simply merge the coefficients
338  if (parent->d == 1) {
339  GFpX eX;
340  for (long i = min(parent->overfield->d, long(v.size())) - 1 ; i >=0 ; i--) {
341  if (!v[i].isZero())
342  SetCoeff(eX, i, v[i].repBase);
343  }
344  parent->overfield->switchContext();
345  e.base = false;
346  e.repBase = 0;
347  conv(e.repExt, eX);
348  e.parent_field = parent->overfield;
349  }
350  // if the element is a scalar
351  else if (scalar) {
352  e.base = false;
353  e.repBase = 0;
354  GFp ec = coeff(rep(v[0].repExt), 0);
355  parent->overfield->switchContext();
356  e.repExt = ec;
357  e.parent_field = parent->overfield;
358  }
359  // the real lift-up algorithm from Section 4
360  else {
361  bool base = parent->d == 1;
362  BigInt p = parent->p;
363  GFpXModulus Q = GFpE::modulus();
364 
365  // take the elements out of v
366  vector<GFpX> W; W.resize(p);
367  for (BigInt i = 0 ; i < p ; i++) {
368  if (i < long(v.size()) && !v[i].isZero()) {
369  if (base) W[i] = v[i].repBase;
370  else W[i] = rep(v[i].repExt);
371  }
372  }
373  // The input lies in GF(p)[x0].
374  // If this extension was built modulo
375  // X^p - X - x0 - 1
376  // this brings the elements into GF(p)[x0+1]
377  if (parent->overfield->plusone) {
378  GFpX xminusone;
379  SetCoeff(xminusone, 1); SetCoeff(xminusone, 0, -1);
380  for (BigInt i = 0 ; i < p ; i++)
381  compose<T>(W[i], W[i], xminusone, p);
382  GFpX tmp;
383  compose<T>(tmp, Q.val(), xminusone, p);
384  build(Q, tmp);
385  }
386 
387  // get the trace form
388  if (Q.tracevec.length() == 0) {
389 #ifdef FAAST_TIMINGS
390  Field<T>::TIME.TRACEVEC = -GetTime();
391 #endif
392  ComputeTraceVec(Q);
393 #ifdef FAAST_TIMINGS
394  Field<T>::TIME.TRACEVEC += GetTime();
395 #endif
396  }
397  GFpX trace; conv(trace, Q.tracevec);
398  // TransposedMul (step 2 of lift-up)
399  TransposedMul<T>(W, Q, trace, p);
400 
401  // if this extension was built modulo
402  // X^p - X - xi^(2p-1)
403  // apply mod* and evaluate*
404  // (steps 2 and 3 of push-down*)
405  if (parent->overfield->twopminusone) {
406  // mod*
407  for (BigInt i = 0 ; i < p ; i++)
408  TransMod<T>(W[i], Q, p);
409  // evaluate*
410  for (BigInt i = 0 ; i < p ; i++)
411  contract<T>(W[i], W[i], 2*long(p) - 1);
412  }
413 
414  // step 4 of push-down*
415  GFpX V;
416  TransPushDownRec<T>(W, V, 0, parent->overfield->d - 1, p);
417 
418  // now get ready to work in the overfield
419  parent->overfield->switchContext();
420 
421  // step 4 of lift-up
422  const GFpXModulus& QQ = GFpE::modulus();
423  GFpX revQ; reverse(revQ, QQ);
424  MulTrunc(V, V, revQ, deg(QQ));
425 
426  // step 5 of lift-up
427  reverse(V, V, deg(QQ) - 1);
428  e.base = false;
429  e.repBase = 0;
430  conv(e.repExt, V);
431  e.parent_field = parent->overfield;
432  const FieldElement<T>& invDiffQQ =
433  parent->overfield->getLiftup();
434  e *= invDiffQQ;
435  }
436  }
