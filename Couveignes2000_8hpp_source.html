23  /* The algorithm ApproximateAS from Section 6 */
24  template <class T> void approximateAS(
25  vector<FieldElement<T> >& v, const FieldElement<T>& tr) {
26  const Field<T>& parent = v[0].parent();
27  for (long j = long(parent.p) - 1 ; j >= 1 ; j--) {
28  FieldElement<T> t = tr;
29  FieldElement<T> binom =
30  parent.primeField().scalar(j);
31  v[j] = v[j-1];
32  for (long h = j + 1 ; h <= long(parent.p) - 1 ; h++) {
33  t *= tr;
34  binom *= parent.primeField().scalar(h);
35  binom /= parent.primeField().scalar(h - j + 1);
36  v[j] -= v[h] * (t * binom);
37  }
38  v[j] /= tr * parent.primeField().scalar(j);
39  v[j] >>= parent;
40  }
41  v[0] = parent.zero();
42  }

44  /* Couveignes' algorithm (Section 6).
45  * Assumes Tr(alpha) = 0
46  */
47  template <class T> void Field<T>::couveignes00(
48  FieldElement<T>& res, const FieldElement<T>& alpha) const {
49 #ifdef FAAST_DEBUG
50  if (alpha.trace() != 0)
51  throw FAASTException("Bad input to couveignes00.");
52 #endif
53  // step 0
54  if (alpha.isZero()) {
55  res = alpha;
56  return;
57  }
58 
59  const Field<T>& parent = *(alpha.parent_field);
60  long i = parent.height;
61  // step 1
62  if (i == 0) {
63  parent.switchContext();
64  const MatGFp& artin = parent.getArtinMatrix();
65 #ifdef FAAST_DEBUG
66  if (parent.stem->artinLine < 0)
67  throw FAASTException("Bad Artin Matrix.");
68 #endif
69  VecGFp low, high;
70  VectorCopy(low, rep(alpha.repExt), parent.stem->artinLine);
71  VectorCopy(high, RightShift(rep(alpha.repExt),
72  parent.stem->artinLine+1),
73  parent.d - 1 - parent.stem->artinLine);
74  append(low, high);
75  // apply the artin matrix
76  VecGFp resV = artin * low;
77  GFpX resX; conv(resX, resV);
78  resX <<= 1;
79  conv(res.repExt, resX);
80  res.repBase = 0;
81  res.base = false;
82  res.parent_field = &parent;
83  return;
84  }
85  // step 2
86  FieldElement<T> eta = alpha;
87  eta.BigPTrace(i-1);
88  // step 3
89  vector<FieldElement<T> > etas;
90  FAAST::pushDown(eta, etas);
91  etas.resize(p);
92 #ifdef FAAST_DEBUG
93  if (etas[long(parent.p)-1] != 0)
94  throw FAASTException("Error in couveignes00.");
95 #endif
96  approximateAS<T>(etas, parent.stem->alpha->trace());
97  // step 4
98  vector<FieldElement<T> > alphas;
99  FAAST::pushDown(alpha, alphas);
100  alphas.resize(1);
101  FieldElement<T> alpha1 = alphas[0];
102  FieldElement<T> gamma = alphas[0].parent_field->one();
103  for (BigInt j = 1 ; j < parent.p ; j++) {
104  gamma *= *(parent.stem->alpha);
105  FieldElement<T> tmp = etas[long(j)];
106  tmp.self_frobenius();
107  tmp *= gamma;
108  alpha1 -= tmp;
109  }
110  // step5
111  couveignes00(etas[0], alpha1);
112  // step 6
113  FAAST::liftUp(etas, res);
114  }

116  /* Return a root in this field of the polynomial
117  * X^p - X - alpha
118  * throws IllegalCoercionException if alpha can't be
119  * coerced to this field
120  * throws IsIrreducibleException if the polynomial has
121  * no roots in this field
122  */
123  template <class T> FieldElement<T>
124  Field<T>::Couveignes2000(const FieldElement<T>& alpha)
125  const throw(IllegalCoercionException, IsIrreducibleException) {
126  if (!isOverFieldOf(*(alpha.parent_field)))
127  throw IllegalCoercionException();
128 
129  // if alpha lies in this level
130  if (alpha.parent_field->stem == stem) {
131  if (alpha.trace() != 0)
132  throw IsIrreducibleException();
133  FieldElement<T> res;
134  couveignes00(res, alpha);
135  return res >> *this;
136  }
137  // if alpha lies one level below
138  else if (alpha.parent_field->stem->overfield == stem) {
139  FieldElement<T> tr = alpha.trace();
140  if (tr == 0) {
141  FieldElement<T> res =
142  stem->subfield->Couveignes2000(alpha);
143  return res >> *this;
144  } else {
145  // simplified version of couveignes00
146  vector<FieldElement<T> > v;
147  v.resize(2);
148  v[1] = tr / stem->alpha->trace();
149  v[0] = *(stem->alpha);
150  v[0] >>= *(alpha.parent_field);
151  v[0] *= -v[1];
152  v[0] += alpha;
153  couveignes00(v[0], v[0]);
154  v[1] >>= *(v[0].parent_field);
155  FieldElement<T> res;
156  FAAST::liftUp(v, res);
157  return res >> *this;
158  }
159  }
160  // else
161  else {
162  FieldElement<T> res =
163  stem->subfield->Couveignes2000(alpha);
164  return res >> *this;
165  }
166  }
