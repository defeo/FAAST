25 /* Half GCD between P and Q, assumes deg P >= deg Q */
26 template <class T> void
27 IterHalfGCD(FieldPolynomial<T>& U0, FieldPolynomial<T>& V0,
28  FieldPolynomial<T>& U1, FieldPolynomial<T>& V1,
29  FieldPolynomial<T>& P, FieldPolynomial<T>& Q,
30  const long d) {
31 
32  U0 = V1 = P.parent().one();
33  U1 = V0 = P.parent().zero();
34 
35  long goal = P.degree() - d;
36 
37  if (Q.degree() <= goal)
38  return;
39 
40  FieldPolynomial<T> q, t;
41 
42  while (Q.degree() > goal) {
43  q.division(P, Q); t.mod(P, Q);
44  P = Q; Q = t;
45 
46  t.product(q, U1); t.negate(); t += U0;
47  U0 = U1; U1 = t;
48 
49  t.product(q, V1); t.negate(); t += V0;
50  V0 = V1; V1 = t;
51  }
52 }

56 /* Half GCD between P and Q, assumes deg P > deg Q */
57 template <class T> void
58 RecHalfGCD(FieldPolynomial<T>& U0, FieldPolynomial<T>& V0,
59  FieldPolynomial<T>& U1, FieldPolynomial<T>& V1,
60  const FieldPolynomial<T>& P, const FieldPolynomial<T>& Q,
61  const long d) {
62 
63  if (Q == 0 || Q.degree() <= P.degree() - d) {
64  U0 = V1 = P.parent().one();
65  U1 = V0 = P.parent().zero();
66 
67  return;
68  }
69 
70  long n = P.degree() - 2*d + 2;
71  if (n < 0) n = 0;
72 
73  FieldPolynomial<T> P1, Q1;
74  P1.RightShift(P, n);
75  Q1.RightShift(Q, n);
76 
77  if (d <= T::consts.HalfGCD_CROSSOVER) {
78  IterHalfGCD<T>(U0, V0, U1, V1, P1, Q1, d);
79  return;
80  }
81  long d1 = (d + 1)/2;
82  if (d1 >= d) d1 = d - 1;
83 
84  FieldPolynomial<T> u0, v0, u1, v1;
85 
86  RecHalfGCD<T>(u0, v0, u1, v1, P1, Q1, d1);
87  // matrix-vector product
88  // |P1| |u0 v0| |P1|
89  // |Q1| <- |u1 v1| |Q1|
90  FieldPolynomial<T> t1, t2;
91  t1.product(u0, P1);
92  t2.product(v0, Q1);
93  t1 += t2;
94  t2.product(u1, P1);
95  Q1 *= v1; Q1 += t2;
96  P1 = t1;
97 
98  long d2 = Q1.degree() - P.degree() + n + d;
99 
100  if (Q1 == 0 || d2 <= 0) {
101  U0 = u0; V0 = v0; U1 = u1; V1 = v1;
102  return;
103  }
104 
105  FieldPolynomial<T> q, y0, w0, y1, w1;
106 
107  q.division(P1, Q1); P1 %= Q1;
108  RecHalfGCD<T>(y0, w0, y1, w1, Q1, P1, d2);
109 
110  // matrix multiplication
111  // |u0 v0| |0 1| |u0 v0|
112  // |u1 v1| <- |1 -q| |u1 v1|
113  q.negate();
114  t1.product(q, u1); t1 += u0;
115  u0 = u1; u1 = t1;
116  t1.product(q, v1); t1 += v0;
117  v0 = v1; v1 = t1;
118 
119  // matrix multiplication (Strassen formula)
120  // |U0 V0| |y0 w0| |u0 v0|
121  // |U1 V1| = |y1 w1| |u1 v1|
122  U0 = V0 = U1 = V1 = P.parent().zero();
123 
124  t1.difference(y0, w0); t1 *= v1;
125  U0 += t1; V0 -= t1;
126 
127  t1.difference(y1, w1); t1 *= u0;
128  U1 += t1; V1 -= t1;
129 
130  t1.sum(u0, u1); t1 *= w1;
131  U0 -= t1; U1 += t1;
132 
133  t1.sum(v0, v1); t1 *= y0;
134  V0 += t1; V1 -= t1;
135 
136  t2.sum(y0, w1); t1.difference(v1, u0); t1 *= t2;
137  U0 -= t1; V1 += t1;
138 
139  t2.sum(y0, y1); t1.sum(u0, v0); t1 *= t2;
140  V1 += t1;
141 
142  t2.sum(w0, w1); t1.sum(u1, v1); t1 *= t2;
143  U0 += t1;
144 }

149  /* GCD between P and Q */
150  template <class T> FieldPolynomial<T>
151  GCD(const FieldPolynomial<T>& P, const FieldPolynomial<T>& Q)
152  throw(NotInSameFieldException) {
153  if (!P.parent_field) return Q;
154  if (!Q.parent_field) return P;
155  P.sameLevel(Q);
156  P.parent_field->switchContext();
157 
158  FieldPolynomial<T> res = P.parent_field->zero();
159  if (P.base) NTL::GCD(res.repBase, P.repBase, Q.repBase);
160  else {
161  FieldPolynomial<T> U, V;
162  XGCD<T>(P, Q, U, V, res);
163  }
164 
165  return res;
166  }

168  /* XGCD between P and Q */
169  template <class T> void
170  XGCD(const FieldPolynomial<T>& P, const FieldPolynomial<T>& Q,
171  FieldPolynomial<T>& U, FieldPolynomial<T>& V, FieldPolynomial<T>& G)
172  throw(NotInSameFieldException) {
173  if (!P.parent_field && !Q.parent_field) G = U = V = P;
174  else if (!P.parent_field) {
175  U = Q.parent_field->zero();
176  V = Q.parent_field->one();
177  G = Q;
178  } else if (!Q.parent_field) {
179  U = P.parent_field->one();
180  V = P.parent_field->zero();
181  G = P;
182  } else {
183  P.sameLevel(Q);
184  FieldPolynomial<T> U1, V1;
185  if (P.degree() >= Q.degree()) {
186  RecHalfGCD<T>(U, V, U1, V1, P, Q, P.degree()+1);
187  } else {
188  RecHalfGCD<T>(V, U, V1, U1, Q, P, Q.degree()+1);
189  }
190 
191  G = U*P + V*Q;
192  // normalize
193  FieldElement<T> lc; G.getCoeff(G.degree(), lc);
194  G /= lc; U /= lc; V /= lc;
195  }
196  }

198  /* Half GCD between P and Q */
199  template <class T> void
200  HalfGCD(FieldPolynomial<T>& U0, FieldPolynomial<T>& V0,
201  FieldPolynomial<T>& U1, FieldPolynomial<T>& V1,
202  const FieldPolynomial<T>& P, const FieldPolynomial<T>& Q,
203  const long d)
204  throw(NotInSameFieldException, BadParametersException){
205  if (d > P.degree() && d > Q.degree())
206  throw BadParametersException("Negative goal degree for HalfGCD");
207  if (d < 0)
208  throw BadParametersException("Negative parameter for HalfGCD");
209 
210  if (!P.parent_field && !Q.parent_field) U0 = V0 = U1 = V1 = P;
211  else if (!P.parent_field) {
212  U0 = V1 = Q.parent_field->zero();
213  U1 = V0 = Q.parent_field->one();
214  } else if (!Q.parent_field) {
215  U1 = V0 = Q.parent_field->zero();
216  U0 = V1 = P.parent_field->one();
217  } else {
218  P.sameLevel(Q);
219 
220  if (P.degree() >= Q.degree()) {
221  RecHalfGCD<T>(U0, V0, U1, V1, P, Q, d);
222  } else {
223  RecHalfGCD<T>(V0, U0, V1, U1, Q, P, d);
224  }
225  }
