23  /* return the least power of p greater than n */
24  long NumPits(const long p, long n) {
25  if (p == 2) return NumBits(n);
26  else {
27  if (n == 0) return 0;
28  if (n < 0) n = -n;
29  long k = long(floor(log(n) / log(p)) + 1);
30  long pk = power_long(p, k);
31  if (pk <= n) {
32  long i = 0;
33  while (pk <= n) { i++; pk *= p; }
34 #if FAAST_DEBUG>=2
35  if (i > 1) cout << "cmath log imprecise by " << i << " p-its." << endl;
36 #endif
37  return k+i;
38  } else {
39  long i = -1;
40  while (pk > n) { i++; pk /= p; }
41 #if FAAST_DEBUG>=2
42  if (i > 1) cout << "cmath log imprecise by " << i << " p-its." << endl;
43 #endif
44  return k - i;
45  }
46  }
47  }

50  template <class T> void expand(typename T::GFpX& res,
51  const typename T::GFpX& P, const long n) {
52  typedef typename T::GFpX GFpX;
53 
54  GFpX restmp;
55  for (long i = deg(P) ; i >= 0 ; i--)
56  SetCoeff(restmp, i*n, coeff(P, i));
57  res = restmp;
58  }

73  /* Store in res the composition Q(R).
74  * Q and R are two polynomials over GF(p)
75  */
76  template <class T> void compose
77  (typename T::GFpX& res, const typename T::GFpX& Q,
78  const typename T::GFpX& R, const typename T::BigInt& p) {
79  typedef typename T::GFpX GFpX;
80 
81  long degree = max(deg(Q),0);
82  long degR = max(deg(R),0);
83  long k = NumPits(p, degree);
84  if (k > 0) {
85  GFpX restmp;
86  long splitdegree = power_long(p, k-1);
87  for (long i = splitdegree * (degree / splitdegree) ; i >= 0 ; i -= splitdegree) {
88  GFpX Q1;
89  SetCoeff(Q1, min(splitdegree - 1, deg(Q) - i)); // hack
90  for (long j = 0 ; j < splitdegree && i+j <= deg(Q) ; j++) {
91  SetCoeff(Q1, j, coeff(Q, i+j));
92  }
93  compose<T>(Q1, Q1, R, p);
94  // Horner's rule
95  GFpX shifted;
96  if (restmp != 0) {
97  for (long h = 0 ; h <= degR ; h++) {
98  if (coeff(R, h) != 0)
99  shifted += coeff(R, h) * LeftShift(restmp, splitdegree*h);
100  }
101  }
102  restmp = Q1 + shifted;
103  }
104  res = restmp;
105  } else {
106  res = Q;
107  }
108  }

142  k += m;

147  ZZ gg;

158  // Deterministic primality test. Only valid if
159  // 17 < n < 341,550,071,728,321
160  bool isJaeschkePrime(long n) {
161  ZZ witn[7];
162  witn[0] = 2; witn[1] = 3; witn[2] = 5;
163  witn[3] = 7; witn[4] = 11; witn[5] = 13;
164  witn[6] = 17;
165 
166  ZZ zzn; zzn = n;
167  for (int i = 0 ; i < 7 ; i++) {
168  if (MillerWitness(zzn, witn[i])) return false;
169  }
170  return true;
171  }

173  /* Inserts n with multiplicity m in the vector */
174  void insertInPlace(vector<pair<long,int> >& factors,
175  long n, int m) {
176  bool inserted = false;
177  vector<pair<long,int> >::iterator it;
178  for (it = factors.begin() ;
179  it != factors.end() && !inserted ; it++) {
180  if (it->first == n) {
181  it->second += m;
182  inserted = true;
183  }
184  if (it->first > n) {
185  factors.insert(it, pair<long,int>(n,m));
186  inserted = true;
187  }
188  }
189  if (!inserted) factors.insert(it, pair<long,int>(n,m));
190  }

192  /* Pollard Rho factorisation algorithm */
193  void factor(long n, vector<pair<long,int> >& factors,
194  const bool trial, const int multiplicity) {
195  int mul;
196  if (trial) {
197  // trial division up to 17
198  long trials[] = {2, 3, 5, 7, 11, 13, 17};
199  for (int i = 0 ; i < 7 && n > 1; i++) {
200  for (mul = 0 ; n % trials[i] == 0 ; mul++) n /= trials[i];
201  if (mul > 0) {
202  insertInPlace(factors, trials[i],
203  mul*multiplicity);
204  }
205  }
206  }
207 
208  if (n <= 1) return;
209  // test primality
210  bool isprime = isJaeschkePrime(n);
211  if (isprime && NumBits(n) > 47) {
212  isprime &= ProbPrime(n, 10);
213  }
214 
215  // if n is prime, simply add it to the factorisation
216  if (isprime) {
217  insertInPlace(factors, n, multiplicity);
218  }
219  // else, use pollard rho until a factor is found
220  else {
221  long fact = 0;
222  while (fact == 0) {
223  long c = 0;
224  while (c == 0) { c = RandomBnd(n); }
225  fact = brent80(n,1,1,c);
226  }
227  for (mul = 0 ; n % fact == 0 ; mul++) n /= fact;
228  factor(fact, factors, false, mul*multiplicity);
229  factor(n, factors, false, multiplicity);
230  }
231  }

233  /* compute the n-th cyclotomic polynomial modulo p */
234  template <class T> void cyclotomic
235  (typename T::GFpX& res, const long n, const typename T::BigInt& p) {
236  typedef typename T::GFpX GFpX;
237 
238  if (n == 1) { SetX(res); SetCoeff(res, 0, -1); }
239  else {
240  vector<pair<long,int> > factors; factor(n, factors);
241  // first process the squarefree factors of n
242  GFpX Phi; cyclotomic<T>(Phi, 1, p);
243  long m = 1; // this variable contains the non-squarefree part
244  vector<pair<long,int> >::iterator it;
245  for (it = factors.begin() ; it != factors.end() ; it++) {
246  m *= power_long(it->first, (it->second - 1));
247  long q = it->first;
248  GFpX Phip; expand<T>(Phip, Phi, q);
249 #ifdef FAAST_DEBUG
250  if (Phip % Phi != 0)
251  throw FAASTException("Error : computing the cyclotomic polynomial.");
252 #endif
253  Phi = Phip / Phi;
254  }
255 
256  // now add the other factors
257  if (m > 1) {
258  expand<T>(res, Phi, m);
259  } else res = Phi;
260  }
261  }
