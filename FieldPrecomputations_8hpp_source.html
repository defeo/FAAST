25  template <class T> typename T::MatGFp artinMatrix
26  (const typename T::BigInt& p, const long line, const typename T::GFpXModulus& P) {
27  typedef typename T::MatGFp MatGFp;
28  typedef typename T::GFpX GFpX;
29 
30  long d = deg(P);
31  // we just want an invertinle d-1 minor
32  MatGFp appl; appl.SetDims(d-1, d-1);
33  // X^p mod P
34  GFpX Xp = PowerXMod(p, P);
35  // column is the column of X^p-X at each iteration
36  GFpX column(0,1);
37  // build the minor
38  for (long i = 1 ; i < d ; i++) {
39  MulMod(column, column, Xp, P);
40  for (long j = 0 ; j < line ; j++) {
41  appl[j][i-1] = coeff(column, j);
42  }
43  for (long j = line+1 ; j < d ; j++) {
44  appl[j-1][i-1] = coeff(column, j);
45  }
46  if (i > line) appl[i-1][i-1]--;
47  else if (i < line) appl[i][i-1]--;
48  }
49  //invert
50  return inv(appl);
51  }

54  template <class T> const FieldElement<T>&
55  Field<T>::getPseudotrace(const long j) const {
56 #ifdef FAAST_DEBUG
57  if (j < 0 || j >= height)
58  throw FAASTException("Bad input to getPseudotrace.");
59 #endif
60 
61  long size = pseudotraces.size();
62  if (size > j) return pseudotraces[j];
63  if (this != stem) return stem->getPseudotrace(j);
64 
65 #ifdef FAAST_TIMINGS
66  TIME.PSEUDOTRACES = -GetTime();
67 #endif
68  pseudotraces.resize(j+1);
69  if (size == 0) {
70  pseudotraces[0] = *alpha;
71  pseudotraces[0].SmallPTrace(baseField().d);
72  size++;
73  }
74  for (long i = size ; i <= j ; i++) {
75  pseudotraces[i] = pseudotraces[i-1];
76  FieldElement<T> t = pseudotraces[i];
77  for (BigInt h = 1 ; h < p ; h++) {
78  t.BigFrob(i-1);
79  pseudotraces[i] += t;
80  }
81  }
82 #ifdef FAAST_TIMINGS
83  TIME.PSEUDOTRACES += GetTime();
84 #endif
85 
86  return pseudotraces[j];
87  }

89  template <class T> const FieldElement<T>& Field<T>::getLiftup() const {
90  if (this != stem) return stem->getLiftup();
91 
92  if ( !(liftuphelper.get()) ) {
93  switchContext();
94 #ifdef FAAST_TIMINGS
95  TIME.LIFTUP = -GetTime();
96 #endif
97  GFpX diffQ; diff(diffQ, GFpE::modulus());
98  FieldElement<T>* helper = new FieldElement<T>();
99  helper->base = false;
100  conv(helper->repExt, diffQ);
101  helper->parent_field = this;
102 
103  helper->self_inv();
104  liftuphelper.reset(helper);
105 #ifdef FAAST_TIMINGS
106  TIME.LIFTUP += GetTime();
107 #endif
108  }
109  return *liftuphelper;
110  }

112  template <class T> const typename
113  T::MatGFp& Field<T>::getArtinMatrix() const {
114  if (this != stem) return stem->getArtinMatrix();
115 
116  if (artinLine == -1 && d > 1) {
117  // We pick a redundant line : it corresponds
118  // to a power of x of trace different from 0.
119  // The residue formula tells us that Tr(x^dep) != 0
120  switchContext();
121  GFpXModulus P = GFpE::modulus();
122 #ifdef FAAST_TIMINGS
123  TIME.ARTINMATRIX = -GetTime();
124 #endif
125  artinLine = d - 1 - deg(diff(P));
126  artin = artinMatrix<T>(p,artinLine,P);
127 #ifdef FAAST_TIMINGS
128  TIME.ARTINMATRIX += GetTime();
129 #endif
130  }
131  return artin;
132  }

145 #ifdef FAAST_TIMINGS

147 #endif
